<script>
function getImages(){
// List of folders in directory
const folders = ["small", "medium", "large"];

// Loop through each folder and grab images
for(let i=0;i<folders.length;i++){
const directory = "images/" + folders[i] + "/";

// Get list of images from the current directory
let fileExtension = ".jpg";
let files = fs.readdirSync(directory);

// Create HTML element with image source and link to page
files.forEach(function (file) {
if (path.extname(file).toLowerCase() === fileExtension) {

let srcPath = directory + file;

let htmlElement = document.createElement('img');
htmlElement.src = srcPath;

// Add click event listener to open page when image is clicked
htmlElement.addEventListener("click", () => window.open('page-to-open?imgSrc='+srcPath));

// Append img element to parent element in HTML page
document.getElementById('parent-element').appendChild(htmlElement);

}
});
}

}

</script>

</head>

<body onload="getImages()"> <!-- Load images when page is rendered--> <div id="parent-element"></div> <!-- Parent element for all images --> </body> </html>
